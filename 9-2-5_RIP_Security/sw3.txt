configure snmp sysname sw3

create vlan Routing1
configure vlan Routing1 tag 1001
configure vlan Routing1 ipaddress 192.168.10.3/24
configure vlan Routing1 add port 12 tagged
enable ipforwarding vlan Routing1

create vlan Classless3
configure vlan Classless3 tag 2003
configure vlan Classless3 ipaddress 30.0.0.1/24
enable loopback-mode vlan Classless3
enable ipforwarding vlan Classless3

enable rip
enable rip export direct cost 1
configure rip add vlan Routing1
configure rip vlan Routing1 txmode v2only
configure rip add vlan Classless3
configure rip vlan Classless3 txmode v2only
