configure snmp sysname sw1

create vlan Routing1
configure vlan Routing1 tag 1001
configure vlan Routing1 ipaddress 192.168.10.2/24
configure vlan Routing1 add port 12 tagged
enable ipforwarding vlan Routing1

create vlan Classless2
configure vlan Classless2 tag 2002
configure vlan Classless2 ipaddress 20.0.0.1/24
enable loopback-mode vlan Classless2
enable ipforwarding vlan Classless2

enable rip
enable rip export direct cost 1
configure rip add vlan Routing1
configure rip vlan Routing1 txmode v2only
configure rip add vlan Classless2
configure rip vlan Classless2 txmode v2only
