configure snmp sysname sw1

create vlan Routing1
configure vlan Routing1 tag 1001
configure vlan Routing1 ipaddress 192.168.10.1/24
configure vlan Routing1 add port 12 tagged
enable ipforwarding vlan Routing1

create vlan Classless1
configure vlan Classless1 tag 2001
configure vlan Classless1 ipaddress 10.0.0.1/24
enable loopback-mode vlan Classless1
enable ipforwarding vlan Classless1

enable rip
enable rip export direct cost 1
configure rip add vlan Routing1
configure rip vlan Routing1 txmode v1only
configure rip add vlan Classless1
configure rip vlan Classless1 txmode v1only
