configure snmp sysname sw3

create vlan Routing2
configure vlan Routing2 tag 1002
configure vlan Routing2 ipaddress 192.168.20.3/24
configure vlan Routing2 add port 12 tagged
enable ipforwarding vlan Routing2

create vlan Routing3
configure vlan Routing3 tag 1003
configure vlan Routing3 ipaddress 192.168.30.3/24
configure vlan Routing3 add port 11 tagged
enable ipforwarding vlan Routing3

create vlan Classless1
configure vlan Classless1 tag 2001
configure vlan Classless1 ipaddress 10.0.0.1/24
enable loopback-mode vlan Classless1
enable ipforwarding vlan Classless1

create vlan Classless2
configure vlan Classless2 tag 2002
configure vlan Classless2 ipaddress 20.0.0.1/24
enable loopback-mode vlan Classless2
enable ipforwarding vlan Classless2

enable rip
enable rip export direct cost 1

configure rip add vlan Routing2
configure rip vlan Routing2 txmode v2only
configure rip add vlan Routing3
configure rip vlan Routing3 txmode v2only
configure rip add vlan Classless1
configure rip vlan Classless2 txmode v2only
configure rip add vlan Classless2
configure rip vlan Classless2 txmode v2only
